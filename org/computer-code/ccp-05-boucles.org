    while True:
	print('Menu')
	print('1: Entrer votre prénom')
	print('2: Entrer votre âge')
	print('q: Quitter')
	choix = input('Votre choix : ')

* Continue

  #+BEGIN_SRC python
    while True:
	print('Quel est votre langage de programmation favori ?')
	nom = input()
	if nom != 'Python':
	    continue
	print('Et oui ! Python est le meilleur langage de programmation !')
	version = int(input('Quelle version utilisez-vous ? '))
	if version == 3:
	    break
	else:
	    print('Il faut vous mettre à jour !')
  #+END_SRC

* La boucle for

  #+BEGIN_SRC python
    for i in range(10):
	print(i)

    for letter in 'hello':
	print(letter)
  #+END_SRC

* Break et continue avec la boucle for

  #+BEGIN_SRC python
    for i in range(10):
	print(i)
	if i > 5:
	    break

    for i in range(10):
	print(i)
	if i < 5:
	    continue
  #+END_SRC
