* L'affectation parallèle

  #+BEGIN_SRC python
    btc = ['premier', 'plus gros réseau', 'plus gros volume']
    anciennete = btc[0]
    reseau = btc[1]
    volume = btc[2]
    print("La même chose en 2 lignes")
    btc = ['premier', 'plus gros réseau', 'plus gros volume']
    anciennete, reseau, volume = btc
    c1, c2 = 'btc', 'eth'
    print("c1 =", c1, "c2 =", c2)
    c2, c1 = c1, c2
    print("c1 =", c1, "c2 =", c2)
  #+END_SRC

* Trier des éléments
  
  #+BEGIN_SRC python
    nombres = [2, 4.5, -9, 5.75, 7, 6]
    nombres.sort()
    print(nombres)
    nombres.sort(reverse=True)
    print(nombres)
    crypto = ['Bitcoin', 'Litecoin', 'Ethereum']
    crypto.sort()
    print(crypto)
    lettres = ['a', 'b', 'A', 'B']
    lettres.sort(key=str.lower)
    print(lettres)
  #+END_SRC
